abs

abs(n);
Returns the absolute value of numeric parameter n

addPlot

addPlot(list,[options]); //adds plot flat numeric list or xy-list
addPlot(xList,yList,[options]); //plots flat numeric list or xy-list
addPlot(yExpression,t0,t1,samples,[options]); //plots yExpression versus t in [t0,t1]
addPlot(xExpression,yExpression,t0,t1,samples,[options]); //plots yExpression versus xExpression for t in [t0,t1]

arccos

arccos(n);
Returns the arccosine of numeric parameter n

arcsin

arcsin(n);
Returns the arcsine of numeric parameter n

arctan

tan(n);
Returns the arctangent of numeric parameter n

argMax

argMax(L);
Returns the index of the greatest element OUT of list L (or the first index if ambiguous)

argMin

argMin(L);
Returns the index of the smallest element OUT of list L (or the first index if ambiguous)

ceil

ceil(x);
Returns the smallest integer >=x

copy

copy(S,start,length):
Returns the substring of S starting at index start and having specified length
Note: Character indexes start with 1!

cos

cos(n);
Returns the cosine of numeric parameter n

execAsync

execAsync(programPath:string,parameters ...);
Starts the specified program and returns true

execSync

execSync(programPath:string,parameters ...);
Executes the specified program and returns the text output

exp

exp(n);
Returns the exponential of numeric parameter n

expression

expression(S:string);
Returns an expression parsed from S

fileContents

fileContents(filename:string);
Returns the contents of the specified file as one string

fileExists

fileExists(filename:string);
Returns true if the specified file exists and false otherwise

fileLines

fileLines(filename:string);
Returns the contents of the specified file as a list of strings
Information on the line breaks is lost

files

files(searchPattern:string);
Returns a list of files matching the given search pattern

flatten

flatten(L,...);
Returns all parameters as a flat list.

floor

floor(x);
Returns the largest integer <=x

folders

folders(searchPattern:string);
Returns a list of folders matching the given search pattern

getPlotAutoscale

getPlotAutoscale;
Returns the current autoscale settings per axis as a tuple of two booleans.

getPlotAxisStyle

getPlotAxisStyle([sx,sy]);
Returns the current axis-style as a tuple of two integers.

getPlotLogscale

getPlotLogscale;
Returns the current log-scale settings per axis as a tuple of two booleans.

getPlotPreserveAspect

getPlotPreserveAspect;
Returns a boolean indicating whether the aspect ratio will be preserverd for the next plot

getPlotRange

getPlotRange;
Returns the plot-range of the last plot as a nested list: [[x0,x1],[y0,y1]]

head

head(L);
Returns the first element of list L or [] if L is empty
head(L,k);
Returns the first min(k,size(L)) elements of L or [] if L is empty

interpret

interpret(filenameOrPackageId:string);
Evaluates the specified file and returns the text output

length

length(S:string);
Returns the number of characters in string S

ln

ln(n);
Returns the natural logarithm of numeric parameter n

lower

lower(S:string);
Returns an lowercase representation of S

max

max(L);
Returns the greatest element OUT of list L
max(x,y,...);
Returns the greatest element OUT of the given parameters

min

min(L);
Returns the smallest element OUT of list L
min(x,y,...);
Returns the smallest element OUT of the given parameters

myPath

returns the path to the current package

not

not(b:boolean);
not(b:booleanList);
Returns the negated value of b
not(i:int);
not(i:intList);
Returns the bitwise negated value of i

plot

plot(list,[options]); //plots flat numeric list or xy-list
plot(xList,yList,[options]); //plots flat numeric list or xy-list
plot(yExpression,t0,t1,samples,[options]); //plots yExpression versus t in [t0,t1]
plot(xExpression,yExpression,t0,t1,samples,[options]); //plots yExpression versus xExpression for t in [t0,t1]

pos

pos(subString,searchInString);
Returns the index of the first occurence of subString in searchInString or 0 if there is none
Note: Character indexes start with 1!

print

print(...);
Prints OUT the given parameters and returns true
if tabs and line breaks are part of the output, a default pretty-printing is used

random

random;
Returns a random value in range [0,1]
random(n);Returns a list of n random values in range [0,1]

renderToFile

renderToFile(filename,width,height,[supersampling]);
Renders the current plot to a file.

replace

replace(source:string,lookFor,replaceBy);
Recursively replaces all occurences of lookFor in source by replaceBy
lookFor and replaceBy may be of type string or stringList

replaceOne

replaceOne(source:string,lookFor,replaceBy);
Replaces the first occurences of lookFor in source by replaceBy
lookFor and replaceBy may be of type string or stringList

round

round(x);
Returns the value of x, rounded to the nearest integer
round(x,k);
Returns the value of x rounded to k-digits precision

setPlotAutoscale

setPlotAutoscale([forX,forY]);
Sets autoscale per axis and returns true
Expects a tuple of two booleans as parameter.

setPlotAxisStyle

setPlotAxisStyle([sx,sy]);
Sets the axis style for the next plot and returns true.

setPlotLogscale

setPlotLogscale([forX,forY]);
Sets log-scale per axis and returns true
Expects a tuple of two booleans as parameter.

setPlotPreserveAspect

setPlotPreserveAspect(b:boolean);
Sets or un-sets preservation of aspect ratio for the next plot.

setPlotRange

setPlotRange([[x0,x1],[y0,y1]]);
Sets the plot-range for the next plot and returns true.

sign

sign(n);
Returns the sign of numeric parameter n

sin

sin(n);
Returns the sine of numeric parameter n

size

size(L);
Returns the number of elements in list L

softCast

softCast(X);
Returns a simplified version of X, trying to parse integers, real values and booleans

sort

sort(L);
Returns list L sorted ascending (using fallbacks for uncomparable types)

sortPerm

sortPerm(L);
Returns indexes I so that L%I==sort(L)

split

split(S:string;splitter:string);
Returns a list of strings obtained by splitting S at the specified splitters
The splitters themselves are not contained in the result

sqr

sqr(n);
Returns the square of numeric parameter n

sqrt

sqrt(n);
Returns the square root of numeric parameter n

string

string(X);
Returns a string-representation of X

tail

tail(L);
Returns list L without the first element
tail(L,k);
Returns L without the first k elements

tan

tan(n);
Returns the tangent of numeric parameter n

time

time(E:expression);
Evaluates E (without parameters) and returns a nested List with evaluation details.

tokenSplit

tokenSplit(S:string);
Returns a list of strings from S

trim

trim(S:string);
Returns string S without leading or trailing spaces

trueCount

trueCount(B:booleanList);
Returns the number of true values in B

unique

unique(L);
Returns list L sorted ascending and without duplicates

upper

upper(S:string);
Returns an uppercase representation of S

writeFile

writeFile(filename:string, content:string);
Writes the specified content to the specified file and returns true

writeFileLines

writeFileLines(filename:string, content:stringList);
Writes the specified content to the specified file (using system-default line breaks) and returns true
